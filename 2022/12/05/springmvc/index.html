<?xml version="1.0" encoding="UTF-8"?>
<web-app xmlns="http://xmlns.jcp.org/xml/ns/javaee"
         xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
         xsi:schemaLocation="http://xmlns.jcp.org/xml/ns/javaee http://xmlns.jcp.org/xml/ns/javaee/web-app_4_0.xsd"
         version="4.0">

    <!--
    前端控制器
    如果我们想要使用springmvc，那么我们工程中的每个请求都要经过前端控制器才可以生效
    -->
    <servlet>
        <servlet-name>springmvc</servlet-name>
        <servlet-class>org.springframework.web.servlet.DispatcherServlet</servlet-class>
        <!-- 配置加载springmvc的配置文件 -->
        <init-param>
            <param-name>contextConfigLocation</param-name>
            <param-value>classpath:springmvc.xml</param-value>
        </init-param>
    </servlet>

    <servlet-mapping>
        <servlet-name>springmvc</servlet-name>
        <url-pattern>/</url-pattern>
    </servlet-mapping>
</web-app>

1.4springmvc底层执行流程

用户发起请求的时候它会经过一个前端控制器叫disparchservlet ，disparchservlet会根据请求走第2步。去找到它的映射器知道它映射了谁。 把这个映射器返回给disparchservlet 再根据这个映射器去适配映射器 适配的这个映射器，说白了就是一个controller 由具体的conctrller去执行。返回一个Model and View 通过Model and View 去配置具体的视图解析器 视图解析器返回给前端调用

1.5参数绑定

1.5.1内置对象

在springmvc中其实很多参数都是可以通过前端控制器传递的，所以有很多默认的内置参数，这些内置参数你要使用它，你就可以在形参中添加它，例如

  • HttpServletRequest
  • HttpSession
  • HttpServletResponse

1.5.1参数绑定实体对象

要求页面的name对应的参数值和实体对象中的属性名称相同，并且实体对应按照规范提供了get/set方法。那么我们从页面传递过来的参数就可以直接与controller方法中的实体对象直接进行绑定。

1.7.6Controller层代码

  • 查询方法
  1. 调用业务层方法查询数据库得到数据
  2. 将数据传到request域中方便jsp获取数据
  3. 页面跳转
  • 上传方法
  1. 在controller中的参数列表调用MultipartFile接口的getOriginalFilename来获取上传文件名称
  2. 利用UUID和上传文件名进行拼接防止文件名重复
  3. 获取文件类型可以根据文件类型进行一些判断
  4. 获取在server.xml中添加的文件上传地址，和文件名进行拼接得到完成长传路径
  5. 利用transferTo进行文件上传
  6. 将文件路径地址保存到数据库中
  7. 设置文件路径到实体对像中
  8. 调用业务层添加方法参数为实体对象
  9. 根据返回值进行页面跳转如果成功进行重定向失败请求转发
package com.woniu.maven.web.controller;

import com.woniu.maven.pojo.Building;
import com.woniu.maven.service.BuildingService;
import org.springframework.stereotype.Controller;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.multipart.MultipartFile;

import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpSession;
import java.io.File;
import java.io.IOException;
import java.util.List;
import java.util.UUID;

@Controller
public class BuildingController {

    private BuildingService buildingService = new BuildingService();

    /*
     * request就是参数绑定中的内置对象，需要用到就写，不用就不写
     */
    @RequestMapping("/selectBuilding")
    public String selectBuilding(HttpServletRequest request) {

        // 调用了业务层
        List<Building> list = buildingService.select();

        // 把结果存储到作用域
        request.setAttribute("list", list);

        // 通过返回的字符串来实现请求转发或者重定向:默认执行的是请求转发
        return "forward:/index.jsp";
    }

    /**
     *
     * @param building
     * @param source : 就是专门用于接收文件上传的文件
     */
    @RequestMapping("/addBuilding")
    public String addBuilding(Building building, MultipartFile source) {

        //一、 将图片文件保存在we把服务器中

        //1. 参数绑定：要求参数名和conroller中的形参名一致就能参数绑定
        //2. 参数绑定：直接绑定对象，要求对象的属性必须得与jsp页面的name的值一一对应
        //3. 参数绑定： 接收图片文件

        //1. 图片的原始名称
        String originalFilename = source.getOriginalFilename();
        //2. 使用uuid随机一个名称和图片的原名称进行拼接
        String uuid = UUID.randomUUID().toString().replace("-", "").toUpperCase();
        //3. 拼接原始名称和uuid随机的名称
        String fileName = uuid + "_" + originalFilename;

        //4. 文件的类型：这个是文件的类型，我们可以控制如果不是图片就不让上传
        String contentType = source.getContentType();

        //5. 获取到img这个目录的真是路径
        String realPath = "C:\\ftp\\imgs"; // img/fileName
        String read_img_path = realPath + "\\" + fileName;

        // 使用multipart对象的方法保存文件
        try {
            //6. 就是将你的上传的文件流保存我们刚才计算的指定路径下
            source.transferTo(new File(read_img_path));
        } catch (IOException e) {
            e.printStackTrace();
        }

        //二、 保存图片的路径到数据 ： jdbc
        //1. 设置图片路径到实体对象
        building.setbImg(fileName);

        //2. 调用业务层添加楼宇
        int code = buildingService.addBuilding(building);

        //3. 跳转
        if(code == 1) {
            return "redirect:/success.jsp"; // 重定向
        }else {
            return "forward:/error.jsp"; // 请求转发
        }
    }
}
